149void connectAdafruitIO() {

151
152 MYDEBUG_PRINT("-AdafruitIO : Connexion au broker ... ");
153 int8_t ret;

155 MYDEBUG_PRINT("[ERREUR : ");
156 MYDEBUG_PRINT(MyAdafruitMqtt.connectErrorString(ret));

158 MyAdafruitMqtt.disconnect(); // Deconnexion pour être propre
159 delay(10000); // On attend 10 secondes avant de retenter le coup
160 }
